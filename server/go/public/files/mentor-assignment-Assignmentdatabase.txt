Table users {
  id integer [primary key]
  firstname varchar
  lastname varchar
  email varchar
  password_hash varchar
  role_id integer
}

Table user_roles{
  role_id integer [primary key]
  role_name varchar
}

Ref: users.role_id > user_roles.role_id 

Table bootcamps{
  id integer [primary key]
  name varchar
  learning_outcomes text
  target_audientce text
}

Table stacks{
  id integer [primary key]
  name varchar
}

Table bootcamps_stacks{
  id integer [primary key]
  bootcamp_id integer
  stack_id integer
}

Ref: bootcamps_stacks.bootcamp_id > bootcamps.id
Ref: bootcamps_stacks.stack_id > stacks.id

Table mentors{
  mentor_id integer [primary key]
   user_id integer
}
Ref: mentors.user_id > users.id
Table teaches{
  id integer [primary key]
  stack_id integer
  mentor_id integer
}

Ref: teaches.stack_id > stacks.id
Ref: teaches.mentor_id > mentors.mentor_id


Table users_bootcamps {
  id integer [primary key]
  user_id integer
  bootcamp_id integer
}

Ref: users_bootcamps.user_id > users.id
Ref: users_bootcamps.bootcamp_id > bootcamps.id

Table students{
  student_id integer [primary key]
}
Ref: students.student_id > users.id
